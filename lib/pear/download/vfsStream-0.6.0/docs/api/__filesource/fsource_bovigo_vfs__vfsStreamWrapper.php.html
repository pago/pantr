  226.         $path          = vfsStream::path($path);
  227.         $this->content = $this->getContentOfType($path, vfsStreamContent::TYPE_FILE);
  228.         if (null !== $this->content) {

  230.                 if (!($options & STREAM_REPORT_ERRORS)) {
  231.                     trigger_error('File ' . $path . ' already exist, can not open with mode x', E_USER_WARNING);
  232.                 }
  233.  
  234.                 return false;
  235.             }
  236.  
  237.             $this->content->seek(0, SEEK_SET);
  238.             if (self::TRUNCATE === $mode) {
  239.                 $this->content->setContent(''); // truncate
  240.             } elseif (self::APPEND === $mode) {
  241.                 $this->content->seek(0, SEEK_END);
  242.             }
  243.  
  244.             return true;
  245.         }
  246.         
  247.         $names = $this->splitPath($path);
  248.         $dir   = $this->getContentOfType($names['dirname'], vfsStreamContent::TYPE_DIR);
  249.         if (null === $dir) {
  250.             if (!($options & STREAM_REPORT_ERRORS)) {
  251.                 trigger_error('Directory ' . $names['dirname'] . ' does not exist', E_USER_WARNING);
  252.             }
  253.  
  254.             return false;
  255.         } elseif ($dir->hasChild($names['basename']) === true) {

  257.                 trigger_error('Directory ' . $names['dirname'] . ' already contains a director named ' . $names['basename'], E_USER_WARNING);
  258.             }
  259.  
  260.             return false;
  261.         }
  262.

  264.             if (!($options & STREAM_REPORT_ERRORS)) {
  265.                 trigger_error('Can not open non-existing file ' . $path . ' for reading', E_USER_WARNING);
  266.             }
  267.  
  268.             return false;
  269.         }

  273.     }

  465.     /**
  466.      * creates a new directory
  467.      *

  472.      */
  473.     public function mkdir($path, $mode, $options)
  474.     {
  475.         $mode = ((null == $mode) ? (0777) : ($mode));
  476.         $path = vfsStream::path($path);
  477.         if (null === self::$root) {
  478.             self::$root = vfsStream::newDirectory($path, $mode);
  479.             return true;
  480.         }
  481.         
  482.         $maxDepth = count(explode('/', $path));
  483.         $names    = $this->splitPath($path);
  484.         $newDirs  = $names['basename'];
  485.         $dir      = null;
  486.         $i        = 0;
  487.         while ($dir === null && $i < $maxDepth) {
  488.             $dir     = $this->getContent($names['dirname']);
  489.             $names   = $this->splitPath($names['dirname']);
  490.             $newDirs = $names['basename'] . '/' . $newDirs;
  491.             $i++;
  492.         }
  493.         
  494.         if (null === $dir || $dir->getType() !== vfsStreamContent::TYPE_DIR) {
  495.             return false;
  496.         }
  497.         
  498.         $newDirs = str_replace($dir->getName() . '/', '', $newDirs);
  499.         $recursive = ((STREAM_MKDIR_RECURSIVE & $options) !== 0) ? (true) : (false);
  500.         if (strpos($newDirs, '/') !== false && false === $recursive) {
  501.             return false;
